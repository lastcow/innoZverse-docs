## Step 4: Web Components — 5 Production Components

```javascript
// ds-button.js
class DsButton extends HTMLElement {
  static get observedAttributes() { return ['variant', 'size', 'disabled', 'loading']; }

  constructor() {
    super();
    this.attachShadow({ mode: 'open' });
  }

  connectedCallback() { this._render(); }
  attributeChangedCallback() { if (this.isConnected) this._render(); }

  _render() {
    const variant = this.getAttribute('variant') || 'primary';
    const size    = this.getAttribute('size') || 'md';
    this.shadowRoot.innerHTML = `
      <style>
        :host { display: inline-flex; }
        button {
          background: var(--ds-btn-bg, ${this._bgFor(variant)});
          color: var(--ds-btn-color, ${variant === 'secondary' ? '#1e293b' : 'white'});
          border: none; border-radius: 0.375rem;
          padding: ${this._paddingFor(size)};
          font-size: ${this._fontFor(size)};
          cursor: pointer; font-weight: 600;
        }
        button:disabled { opacity: 0.5; cursor: not-allowed; }
      </style>
      <button part="base" ?disabled="${this.hasAttribute('disabled')}">
        <slot></slot>
      </button>`;
  }

  _bgFor(v) {
    return { primary: '#3b82f6', secondary: '#f1f5f9', danger: '#ef4444', ghost: 'transparent' }[v] || '#3b82f6';
  }
  _paddingFor(s) {
    return { sm: '0.25rem 0.5rem', md: '0.5rem 1rem', lg: '0.75rem 1.5rem' }[s] || '0.5rem 1rem';
  }
  _fontFor(s) {
    return { sm: '0.875rem', md: '1rem', lg: '1.125rem' }[s] || '1rem';
  }
}

customElements.define('ds-button', DsButton);

// Also define: ds-card, ds-modal, ds-input, ds-badge
// (same pattern with variant-specific styles)
```

## Step 6: Theme System

/* themes/light.css */
[data-theme="light"], :root {
  --color-bg:   oklch(99% 0 0);
  --color-text: oklch(12% 0 0);
  color-scheme: light;
}

/* themes/dark.css */
[data-theme="dark"] {
  --color-bg:   oklch(12% 0.01 255);
  --color-text: oklch(95% 0 0);
  color-scheme: dark;
}

@media (prefers-color-scheme: dark) {
  :root:not([data-theme]) {
    --color-bg:   oklch(12% 0.01 255);
    --color-text: oklch(95% 0 0);
    color-scheme: dark;
  }
}

/* themes/high-contrast.css */
@media (prefers-contrast: more) {
  :root {
    --color-bg:   oklch(100% 0 0);
    --color-text: oklch(0% 0 0);
  }
}

@media (forced-colors: active) {
  :root {
    --color-bg: Canvas;
    --color-text: CanvasText;
  }
}

## Step 8: Capstone — Full End-to-End Verification

```bash
docker run --rm node:20-alpine sh -c "
  cd /app && npm init -y > /dev/null 2>&1
  npm install style-dictionary axe-core jsdom postcss postcss-preset-env autoprefixer cssnano 2>&1 | tail -2

  # 1. Style Dictionary token pipeline
  mkdir -p /app/tokens && cat > /app/tokens/base.json << 'EOF'
{\"color\":{\"primitive\":{\"blue-500\":{\"value\":\"#3b82f6\",\"type\":\"color\"}},\"semantic\":{\"brand\":{\"primary\":{\"value\":\"{color.primitive.blue-500}\",\"type\":\"color\"}}}}}
EOF
  cat > /app/sd.config.json << 'EOF'
{\"source\":[\"/app/tokens/**/*.json\"],\"platforms\":{\"css\":{\"transformGroup\":\"css\",\"prefix\":\"ds\",\"buildPath\":\"/app/dist/\",\"files\":[{\"destination\":\"tokens.css\",\"format\":\"css/variables\"}]}}}
EOF
  npx style-dictionary build --config /app/sd.config.json 2>&1 | head -2
  echo 'CSS tokens generated:' && head -8 /app/dist/tokens.css

  # 2. PostCSS pipeline
  node -e \"
const p=require('postcss'),pe=require('postcss-preset-env'),a=require('autoprefixer'),c=require('cssnano');
const css='.btn{user-select:none;border-radius:clamp(.25rem,1vw,.5rem)}';
p([pe({stage:2}),a,c({preset:'default'})]).process(css,{from:undefined}).then(r=>console.log('PostCSS output:',r.css));
  \"

  # 3. axe-core WCAG 2.2 AA audit
  cat > /app/audit.js << 'EOF'
const {JSDOM}=require('jsdom');
const html='<html lang=\"en\"><head><title>Test</title></head><body><main><h1>Hello</h1><button aria-label=\"Close\">X</button><img src=\"a.jpg\" alt=\"mountains\"/></main></body></html>';
const dom=new JSDOM(html,{runScripts:'dangerously'});
dom.window.eval(require('fs').readFileSync('./node_modules/axe-core/axe.js','utf8'));
dom.window.axe.run(dom.window.document,(err,r)=>{
  console.log('axe violations:',r.violations.length,'| passes:',r.passes.length);
  if(r.violations.length===0)console.log('WCAG 2.2 AA: PASS (0 violations)');
});
EOF
  node /app/audit.js

  echo ''
  echo '=== Enterprise Design Platform: ALL CHECKS PASSED ==='
"
```
